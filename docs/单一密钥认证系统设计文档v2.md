# 单一密钥认证系统设计文档 v2.0

## 1. 系统概述

### 1.1 设计目标
实现基于单一API密钥的认证系统，完全替代用户名密码登录方式，提供简洁、安全、易管理的系统间认证机制。

### 1.2 核心特性
- **单一密钥**: 一个auth_key字段实现身份识别和认证
- **租户绑定**: 密钥格式包含租户信息，支持多租户架构
- **简洁设计**: 最小化字段设计，降低复杂度
- **无缝集成**: 与现有fan-db-ui认证体系完全兼容
- **智能重试**: 认证失败自动重试，提升用户体验

### 1.3 适用场景
- 系统间API调用认证
- 前端应用的租户认证
- 内部服务认证
- 开发和测试环境快速认证

## 2. 数据库设计

### 2.1 字段设计
在现有用户表(`users`)中新增以下字段：

```sql
-- 用户认证密钥表字段扩展
ALTER TABLE users ADD COLUMN auth_key VARCHAR(64) UNIQUE;
ALTER TABLE users ADD COLUMN key_status ENUM('active', 'disabled') DEFAULT 'active';
```

### 2.2 字段说明
| 字段名 | 类型 | 说明 | 约束 |
|--------|------|------|------|
| `auth_key` | VARCHAR(64) | 用户认证密钥，包含租户信息 | UNIQUE, 可为空 |
| `key_status` | ENUM | 密钥状态：active(活跃)/disabled(禁用) | 默认active |

### 2.3 密钥格式规范
- **格式**: `fandb_team{team_id}_{random_16_chars}`
- **示例**: `fandb_team24_a1b2c3d4e5f6g7h8`
- **长度**: 总长度约32字符
- **组成**: 
  - `fandb`: 系统标识前缀
  - `team{team_id}`: 租户标识
  - `{random_16_chars}`: 16位随机字符串

### 2.4 密钥生成算法
```python
import secrets
import string

def generate_auth_key(team_id: int) -> str:
    """生成认证密钥"""
    # 生成16位随机字符串（数字+字母）
    random_chars = ''.join(secrets.choice(string.ascii_lowercase + string.digits) for _ in range(16))
    return f"fandb_team{team_id}_{random_chars}"

# 示例输出: fandb_team24_a1b2c3d4e5f6g7h8
```

### 4.3 密钥服务
```python
# apps/user/services/auth_key_service.py
import secrets
import string
from typing import Optional
from ..models import User

class AuthKeyService:
    @staticmethod
    def generate_auth_key(team_id: int) -> str:
        """生成认证密钥"""
        random_chars = ''.join(
            secrets.choice(string.ascii_lowercase + string.digits) 
            for _ in range(16)
        )
        return f"fandb_team{team_id}_{random_chars}"
    
    @staticmethod
    def extract_team_id(auth_key: str) -> Optional[int]:
        """从认证密钥中提取team_id"""
        try:
            if auth_key.startswith('fandb_team'):
                team_part = auth_key.split('_')[1]  # team24
                return int(team_part.replace('team', ''))
        except (IndexError, ValueError):
            pass
        return None
    
    @staticmethod
    async def validate_auth_key(auth_key: str) -> Optional[User]:
        """验证认证密钥"""
        user = await get_user_by_auth_key(auth_key)
        if not user:
            return None
        
        if user.key_status != 'active':
            return None
        
        # 验证密钥格式中的team_id与用户team_id一致
        key_team_id = AuthKeyService.extract_team_id(auth_key)
        if key_team_id != user.team_id:
            return None
        
        return user
```

### 4.4 认证接口实现
```python
# apps/auth/auth_key_login.py
from fastapi import APIRouter, HTTPException, Depends
from ..schemas import AuthKeyLoginRequest, AuthKeyLoginResponse
from ..services.auth_key_service import AuthKeyService
from ..common.security import create_access_token

router = APIRouter()

@router.post("/api-key-login", response_model=AuthKeyLoginResponse)
async def api_key_login(request: AuthKeyLoginRequest):
    """基于认证密钥登录"""
    user = await AuthKeyService.validate_auth_key(request.auth_key)
    
    if not user:
        raise HTTPException(
            status_code=401, 
            detail={
                "error": "invalid_auth_key",
                "message": "认证密钥无效或已禁用",
                "code": 40001
            }
        )
    
    # 生成JWT token
    access_token = create_access_token(
        data={"sub": str(user.id), "team_id": user.team_id}
    )
    
    return AuthKeyLoginResponse(
        access_token=access_token,
        token_type="bearer",
        expires_in=3600,
        user_info={
            "id": user.id,
            "username": user.username,
            "team_id": user.team_id,
            "project_id": user.project_id
        }
    )
```

        headers: { 'Content-Type': 'application/json' },
        timeout: 10000
      })
      
      // 提取token
      const token = extractTokenFromResponse(resp.data)
      
      if (token) {
        tokenStorage.set(token)
        authenticatingPromise = null
        return token
      } else {
        throw new Error('AUTH_TOKEN_NOT_FOUND')
      }
    } catch (error) {
      authenticatingPromise = null
      
      // 根据错误类型抛出特定错误
      if (error.response?.status === 401) {
        throw new Error('AUTH_KEY_INVALID')
      } else if (error.code === 'ECONNREFUSED') {
        throw new Error('AUTH_SERVER_UNREACHABLE')
      } else {
        throw error
      }
    }
  })()
  
  return authenticatingPromise
}

// 构建认证密钥登录URL
const buildAuthKeyLoginUrl = (baseUrl: string): string => {
  const cleanBaseUrl = baseUrl.replace(/\/+$/, '')
  return `${cleanBaseUrl}/auth/api-key-login`
}

// Token保证机制
export const ensureToken = async (options: LoginOptions): Promise<string> => {
  // 1. 检查本地存储的token
  const local = tokenStorage.get()
  if (local) return local

  // 2. 检查环境变量中的默认token（兼容性）
  if (ENV_DEFAULT_TOKEN) {
    tokenStorage.set(ENV_DEFAULT_TOKEN)
    return ENV_DEFAULT_TOKEN
  }
  
  // 3. 执行认证密钥登录
  const token = await performAuthKeyLogin(options)
  return token
}

// 认证失败处理（智能重试机制）
let retryCount = 0
const MAX_RETRY_COUNT = 3

export const handleAuthFailure = async (options: LoginOptions): Promise<void> => {
  console.log('检测到认证失败，清除缓存并重新进行认证')
  tokenStorage.clear()
  
  retryCount++
  
  try {
    await performAuthKeyLogin(options)
    console.log('认证密钥重新认证成功')
    retryCount = 0  // 重置重试计数
  } catch (error) {
    console.error(`认证密钥重新认证失败 (第${retryCount}次):`, error)
    
    if (retryCount >= MAX_RETRY_COUNT) {
      // 达到最大重试次数，跳转到错误页面
      console.error('达到最大重试次数，跳转到认证失败页面')
      redirectToAuthFailurePage(error)
      throw new Error('AUTH_MAX_RETRY_EXCEEDED')
    } else {
      throw new Error(`AUTH_RETRY_${retryCount}`)
    }
  }
}

// 跳转到认证失败页面
const redirectToAuthFailurePage = (error: any): void => {
  const errorType = error.message || 'UNKNOWN_ERROR'
  const errorParams = new URLSearchParams({
    type: errorType,
    auth_key: AUTH_KEY ? `${AUTH_KEY.substring(0, 15)}...` : 'not_configured',
    retry_count: retryCount.toString()
  })
  
  // 跳转到错误页面
  window.location.href = `/auth-failure?${errorParams.toString()}`
}

// Token提取函数（保持兼容）
const extractTokenFromResponse = (data: any): string | null => {
  return data?.access_token || data?.token || data?.data?.access_token || null
}

// 业务认证失败检测（保持兼容）
const isAuthFailureBusinessCode = (data: any): boolean => {
  return data?.code === 4003 || data?.error === 'invalid_auth_key'
}

// 验证认证密钥格式
export const validateAuthKeyFormat = (authKey: string): boolean => {
  const pattern = /^fandb_team\d+_[a-z0-9]{16}$/
  return pattern.test(authKey)
}

// 从认证密钥中提取team_id
export const extractTeamIdFromAuthKey = (authKey: string): number | null => {
  const match = authKey.match(/^fandb_team(\d+)_/)
  return match ? parseInt(match[1]) : null
}

# scripts/generate_auth_keys.py
"""
为现有用户批量生成认证密钥的工具脚本
"""
import asyncio
import secrets
import string
from sqlalchemy import select, update
from apps.user.models import User
from db.database import async_session

async def generate_auth_key(team_id: int) -> str:
    """生成认证密钥"""
    random_chars = ''.join(
        secrets.choice(string.ascii_lowercase + string.digits) 
        for _ in range(16)
    )
    return f"fandb_team{team_id}_{random_chars}"

async def generate_auth_keys_for_users():
    """为所有没有认证密钥的用户生成密钥"""
    async with async_session() as session:
        # 查询需要生成密钥的用户
        result = await session.execute(
            select(User).where(
                User.auth_key.is_(None),
                User.team_id.isnot(None)
            )
        )
        users = result.scalars().all()
        
        print(f"找到 {len(users)} 个需要生成认证密钥的用户")
        
        for user in users:
            # 生成认证密钥
            auth_key = await generate_auth_key(user.team_id)
            
            # 更新用户记录
            await session.execute(
                update(User)
                .where(User.id == user.id)
                .values(
                    auth_key=auth_key,
                    key_status='active'
                )
            )
            
            print(f"用户 {user.username} (ID: {user.id}) 生成密钥: {auth_key}")
        
        await session.commit()
        print(f"批量生成完成，共处理 {len(users)} 个用户")

if __name__ == "__main__":
    asyncio.run(generate_auth_keys_for_users())
